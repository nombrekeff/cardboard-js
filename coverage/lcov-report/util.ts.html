/** Receives a function, and returns just the body of the function as a string */
export function justFnBody(fn: Function) {
  let fnStr = fn.toString();
  fnStr = fnStr.replace(/^(.*{)/, '');
  fnStr = fnStr.replace(/}$/, '');
  fnStr = fnStr.replace(/^\(.*\)\s?=>\s?{/, '');
  return fnStr.trim();
}
